Library Integration

Framework-agnostic Libraries

Use <lifecycle> tag

import Map from "map";

<lifecycle
  onMount() {
    this.map = Map();
  }
  onUpdate() {
    this.map.doSomething();
  }
  onDestroy() {
    this.map.destroy();
  }>
